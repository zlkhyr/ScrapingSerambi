def clean_tweet(tweet):
    tweet=re.sub('@[A-Za-z0-9]+','',tweet) #menghapus @
    tweet=re.sub(r'(?:\@|http?\://|https?\://|www)\S+','',tweet) #remove link http
    tweet=' '.join(tweet.split())
    tweet=''.join(c for c in tweet if c not in emoji.UNICODE_EMOJI) #remove emoji
    tweet=tweet.replace('#','').replace('_',' ') #remove hashtag tapi text pada hashtag tetap
    return tweet
In [16]:
def noise_removal(words):
    words=words.translate(str.maketrans('','',string.punctuation+string.digits)) #maketrans untuk memetakan atau mengganti nilai
    words=words.strip()
    return words

def replace_slang_word(words):
    for index in range(len(words)-1):
        index_slang=indo_slang_word.slang==words[index]
        formal=list(set(indo_slang_word[index_slang].formal))
        if len(formal)==1:
            words[index]=formal[0]
    return words
